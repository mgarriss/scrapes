  def start_server
    @server = WEBrick::HTTPServer.new :Port=>4270, :Logger=>Logger.new(nil),
      :DocumentRoot=>File.expand_path('test/public'), :AccessLog=>[]
    @server_thread = Thread.new { @server.start }
  end
